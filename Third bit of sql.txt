Find Duplicate Records
SELECT name, COUNT(*)
FROM employees
GROUP BY name
HAVING COUNT(*) > 1;
Find Second Highest Salary
SELECT MAX(salary)
FROM employees
WHERE salary <
(
   SELECT MAX(salary)
   FROM employees
);
Remove Duplicate Rows
DELETE FROM employees
WHERE id NOT IN
(
   SELECT MIN(id)
   FROM employees
   GROUP BY name
);
